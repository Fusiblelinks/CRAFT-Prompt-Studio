    <script>
        let currentSection = 'C';
        let values = {};
        
        let variantsData = []; 
        let activeVariantIndex = 0;

        const sections = [
            {
                key: "C", label: "Context", color: "bg-cyan-600", soft: "bg-cyan-50/50",
                tagline: "Set the business scenario and core software environment.",
                fields: [
                    { key: "ctx_category", label: "Task Category", type: "select", options: ["Email drafting","Excel / spreadsheet work","Salesforce Lead/Case mgmt","Data entry & cleaning","Report writing","Meeting notes","Process documentation","Customer comms","Graphic Design / PDF","Other"] },
                    { key: "ctx_tool", label: "Software / Tool", type: "select", options: ["Salesforce CRM","Microsoft Excel","Microsoft Word","Microsoft PowerPoint","Microsoft Outlook","Google Docs","Google Sheets","Google Slides","Gmail","Adobe Photoshop","Adobe Acrobat","Notion","Slack","No specific tool"] },
                    { key: "ctx_background", label: "Detailed Background", type: "textarea", placeholder: "What is the problem you're solving or the data you're working with?" }
                ]
            },
            {
                key: "R", label: "Role", color: "bg-emerald-600", soft: "bg-emerald-50/50",
                tagline: "Define the expert persona and desired professional tone.",
                fields: [
                    { key: "role_persona", label: "AI Persona", type: "select", options: ["Administrative Assistant","Salesforce Admin","Excel expert","CRM Specialist","Business writing specialist","Data analyst","HR administrator"] },
                    { key: "role_tone", label: "Communication Tone", type: "select", options: ["Formal & professional","Friendly & approachable","Concise & direct","Authoritative","Diplomatic"] }
                ]
            },
            {
                key: "A", label: "Action", color: "bg-orange-600", soft: "bg-orange-50/50",
                tagline: "The core deliverable and specific task instructions.",
                fields: [
                    { key: "action_type", label: "Primary Action", type: "select", options: ["Write / draft from scratch","Rewrite / improve text","Summarise document","Create Excel formula / SFDC Validation","Clean / reformat data","Analyse CRM metrics"] },
                    { key: "action_detail", label: "Specific Requirements", type: "textarea", placeholder: "List any specific steps, constraints, or unique elements needed." }
                ]
            },
            {
                key: "F", label: "Format", color: "bg-violet-600", soft: "bg-violet-50/50",
                tagline: "Structure, technical format, and length requirements.",
                fields: [
                    { key: "fmt_type", label: "Output Format", type: "select", options: ["Email","Bullet points","Step-by-step guide","Table (Markdown)","Generate Template/File","Excel Spreadsheet (.csv)","Word Document (.doc)","PowerPoint Slides (.ppt)","Data Export (.json)"] },
                    { key: "fmt_length", label: "Output Length", type: "select", options: ["Brief (< 100 words)","Standard (250-500 words)","Detailed (500+ words)"] }
                ]
            },
            {
                key: "T", label: "Target", color: "bg-sky-700", soft: "bg-sky-50/50",
                tagline: "Identify the recipient and their technical proficiency.",
                fields: [
                    { key: "tgt_recipient", label: "Primary Audience", type: "select", options: ["Senior executive","Direct manager","Colleague / peer","External client / Lead","Vendor / supplier"] },
                    { key: "tgt_skill", label: "Tech Proficiency", type: "select", options: ["Non-technical","Standard office user","Salesforce Power user"] }
                ]
            }
        ];

        function init() {
            renderTabs();
            renderSection();
        }

        function renderTabs() {
            const container = document.getElementById('section-tabs');
            container.innerHTML = sections.map(s => `
                <button onclick="setSection('${s.key}')" class="section-tab-btn px-4 py-2 rounded-md text-[10px] font-black uppercase transition-all whitespace-nowrap ${currentSection === s.key ? 'bg-white text-cyan-600 shadow-sm border border-slate-100' : 'text-slate-400'}" id="tab-${s.key}">
                    ${s.key}. ${s.label}
                </button>
            `).join('');
        }

        function renderSection() {
            const section = sections.find(s => s.key === currentSection);
            const container = document.getElementById('section-container');
            const fieldsArea = document.getElementById('fields-area');
            
            container.className = `p-5 min-h-[400px] transition-colors duration-300 ${section.soft}`;
            document.getElementById('section-badge').className = `w-6 h-6 rounded flex items-center justify-center text-white text-[10px] font-black ${section.color}`;
            document.getElementById('section-badge').textContent = section.key;
            document.getElementById('section-title').textContent = section.label;
            document.getElementById('section-tagline').textContent = section.tagline;

            document.querySelectorAll('.section-tab-btn').forEach(btn => {
                btn.className = `section-tab-btn px-4 py-2 rounded-md text-[10px] font-black uppercase transition-all whitespace-nowrap ${btn.id === 'tab-' + currentSection ? 'bg-white text-cyan-600 shadow-sm border border-slate-100' : 'text-slate-400'}`;
            });

            fieldsArea.innerHTML = section.fields.map(f => {
                const val = values[f.key] || "";
                return `
                    <div class="space-y-1.5 animate-fade-in">
                        <label class="text-[10px] font-black text-slate-500 uppercase tracking-widest">${f.label}</label>
                        ${f.type === 'select' ? `
                            <select onchange="updateValue('${f.key}', this.value)" class="w-full h-11 px-3 rounded-md border border-slate-200 bg-white text-sm focus:ring-2 focus:ring-cyan-500 outline-none shadow-sm">
                                <option value="">Select option...</option>
                                ${f.options.map(o => `<option value="${o}" ${val === o ? 'selected' : ''}>${o}</option>`).join('')}
                            </select>
                        ` : `
                            <textarea oninput="updateValue('${f.key}', this.value)" rows="5" placeholder="${f.placeholder}" class="w-full p-3 rounded-md border border-slate-200 bg-white text-sm focus:ring-2 focus:ring-cyan-500 outline-none shadow-sm leading-relaxed">${val}</textarea>
                        `}
                    </div>
                `;
            }).join('');

            const idx = sections.findIndex(s => s.key === currentSection);
            document.getElementById('prev-btn').disabled = idx === 0;
            document.getElementById('prev-btn').onclick = () => setSection(sections[idx - 1].key);
            
            const nextBtn = document.getElementById('next-btn');
            if (idx === sections.length - 1) {
                nextBtn.textContent = "Generate Blueprint";
                nextBtn.className = "inline-flex items-center justify-center rounded-md text-sm font-black uppercase tracking-widest bg-cyan-600 text-white hover:bg-cyan-700 h-11 px-4 py-2 flex-1 ml-4 shadow-md transition-all";
                nextBtn.onclick = handleFinalAction;
            } else {
                nextBtn.textContent = "Next Step";
                nextBtn.className = "inline-flex items-center justify-center rounded-md text-sm font-bold bg-slate-900 text-white hover:bg-slate-800 h-11 px-4 py-2 flex-1 ml-4 transition-all";
                nextBtn.onclick = () => setSection(sections[idx + 1].key);
            }
        }

        function setSection(key) { currentSection = key; renderSection(); }

        function updateValue(key, val) { values[key] = val; updateProgress(); }

        function updateProgress() {
            let filled = 0; let total = 0;
            sections.forEach(s => s.fields.forEach(f => { total++; if (values[f.key]) filled++; }));
            const p = Math.round((filled / total) * 100);
            document.getElementById('progress-text').textContent = p + '%';
            document.getElementById('progress-fill').style.width = p + '%';
        }

        function buildPrompt(variationIndex = 0) {
            const variationPrompts = [
                "Focus on strict professional standards, accuracy, and clear business alignment.",
                "Adopt a highly engaging, expressive, and creatively optimized perspective.",
                "Prioritize maximum efficiency, extreme brevity, and strict formatting with zero conversational fluff."
            ];
            
            const isFile = (values.fmt_type || "").includes("(") || values.fmt_type === "Generate Template/File";
            const fileHint = isFile ? `IMPORTANT: Provide the output purely as raw content for a ${values.fmt_type} file. No conversational conversational preamble.` : "";

            const core = [
                `Context: ${values.ctx_category || 'General task'} using ${values.ctx_tool || 'productivity software'}.`,
                `Background: ${values.ctx_background || 'No background provided'}.`,
                `Role: ${values.role_persona || 'Expert Assistant'} with a ${values.role_tone || 'professional'} tone.`,
                `Objective: ${values.action_type || 'Drafting'}. ${values.action_detail || 'Produce quality content'}.`,
                `Format: ${values.fmt_type || 'Standard'} / ${values.fmt_length || 'Standard length'}.`,
                `Recipient: ${values.tgt_recipient || 'Colleague'} (${values.tgt_skill || 'Standard skills'}).`,
                `Variation Direction: ${variationPrompts[variationIndex]}`,
                fileHint
            ].filter(l => l !== "");

            return `CRAFT Office Studio Brief:\n\n${core.join("\n")}\n\nDeliver the final output based on these constraints.`;
        }

        function handleFinalAction() {
            document.getElementById('empty-state').classList.add('hidden');
            
            variantsData = [
                { title: "V1: Professional Focus", prompt: buildPrompt(0) },
                { title: "V2: Creative Focus", prompt: buildPrompt(1) },
                { title: "V3: Concise Focus", prompt: buildPrompt(2) }
            ];
            activeVariantIndex = 0;
            renderResult();
        }

        function renderResult() {
            const data = variantsData[activeVariantIndex];
            if (!data) return;

            document.getElementById('prompt-code').textContent = data.prompt;
            document.getElementById('active-variant-badge').textContent = data.title;
            document.getElementById('output-content').classList.remove('hidden');
            
            const nav = document.getElementById('variant-nav');
            nav.classList.remove('hidden');
            document.querySelectorAll('.v-tab').forEach((t, i) => {
                t.className = `v-tab px-3 py-1 rounded text-[10px] font-black uppercase transition-all ${i === activeVariantIndex ? 'bg-cyan-600 text-white shadow-sm' : 'text-slate-500 hover:bg-slate-200'}`;
            });
        }

        function switchVariant(idx) {
            activeVariantIndex = idx;
            renderResult();
        }

        // Safe Fallback Clipboard Copier
        function fallbackCopy(text) {
            const textArea = document.createElement("textarea");
            textArea.value = text;
            
            // Avoid scrolling to bottom
            textArea.style.top = "0";
            textArea.style.left = "0";
            textArea.style.position = "fixed";
            textArea.style.opacity = "0";

            document.body.appendChild(textArea);
            textArea.focus();
            textArea.select();

            let successful = false;
            try {
                successful = document.execCommand('copy');
            } catch (err) {
                console.error('Fallback copy method failed', err);
            }

            document.body.removeChild(textArea);
            return successful;
        }

        function copyPrompt() {
            const data = variantsData[activeVariantIndex];
            if (!data) return;
            
            const copied = fallbackCopy(data.prompt);
            if (copied) {
                const btn = document.getElementById('copy-btn');
                const originalText = btn.textContent;
                btn.textContent = "Copied Blueprint!";
                setTimeout(() => btn.textContent = originalText, 2000);
            }
        }

        function handleLaunch() {
            const data = variantsData[activeVariantIndex];
            const p = data ? data.prompt : buildPrompt(0);
            
            fallbackCopy(p);
            window.open("https://gemini.google.com/app", "_blank");
        }

        window.onload = init;
